Bash Reference

pwd

Print current working directory

pwd

ls

List files and directories

ls -la

cd

Change directory

cd /home/project

mkdir

Create directories

mkdir -p app/config/dev

touch

Create empty files

touch test.txt

rm

Remove files/directories

rm -rf temp/
Search & Filtering

grep

Search text patterns

grep "ERROR" app.log

find

Find files and directories

find . -name "*.sh"
Text Processing

awk

Field-based text processing

awk '{print $1}' file.txt

sed

Stream editor and replacements

sed 's/hello/world/' file.txt

cut

Extract columns

cut -d ':' -f1 /etc/passwd
Variables & Input

Variables

Store values

name="Praveen"
echo "$name"

read

Read user input or parse lines

while read -r status file
Conditions & Loops

if

Conditional execution

if [[ -f test.txt ]]; then
  echo "Exists"
fi

case

Pattern matching

case "$status" in
  A) echo "Added" ;;
esac
Pipes & Redirection

Pipe |

Pass output between commands

cat app.log | grep ERROR

Redirect >

Overwrite file output

echo "hello" > file.txt

Show repository status

git status

git fetch

Fetch remote changes

git log

gh pr create

Create pull request using GitHub CLI

gh pr create --base main --head feature-branch

View container logs

docker exec
