Navigation Commands

pwd

Print current working directory

pwd

ls

List files and directories

ls -la

cd

Change directory

cd /home/project
File Commands

mkdir

Create directories

mkdir -p app/config/dev

touch

Create empty files

touch test.txt

rm

Remove files/directories

rm -rf temp/
Search & Filtering

grep

Search text patterns

grep "ERROR" app.log

find

Find files and directories

find . -name "*.sh"
Text Processing

awk

Field-based text processing

awk '{print $1}' file.txt

sed

Stream editor and replacements

sed 's/hello/world/' file.txt

cut

Extract columns

cut -d ':' -f1 /etc/passwd
Variables & Input

Variables

Store values

name="Praveen"
echo "$name"

read

Read user input or parse lines

while read -r status file
Conditions & Loops

if

Conditional execution

if [[ -f test.txt ]]; then
  echo "Exists"
fi

case

Pattern matching

case "$status" in
  A) echo "Added" ;;
esac
Pipes & Redirection

Pipe |

Pass output between commands

cat app.log | grep ERROR

Redirect >

Overwrite file output

echo "hello" > file.txt
Git Commands for CI/CD

git diff

Compare changes

git diff --name-status origin/main...HEAD

git fetch

Fetch remote changes

git fetch origin main
CI/CD Bash Patterns

Read git diff

Parse changed files

while read -r status file
 do
   echo "$status -> $file"
 done <<< "$changed_files"

Read config file

Read key-value config

while IFS='=' read -r pattern severity
 do
   echo "$pattern -> $severity"
 done < severity.conf

Fail pipeline

Stop workflow intentionally

exit 1
